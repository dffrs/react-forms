    function getByPattern(obj: Record<string, string>, pattern: string) {
      const regex = new RegExp(
        '^' + pattern.replace(/\./g, '\\.').replace(/\*/g, '[^.]+') + '$'
      );
    
      return Object.entries(obj).find(([key]) => regex.test(key))?.[1];
    }
